\begin{tikzpicture}
		[tdplot_main_coords, scale=1.5,
    cube/.style={fill=bl!25!bg, draw=fg, thick, fill opacity = 0.8},
    force/.style={draw=re, thick, -stealth}]

  \filldraw[cube] (0,0,0) -- ++(0,1,0) -- ++(2,0,0) -- ++(0,-1,0) -- cycle;
  \filldraw[cube] (0,0,0) -- ++(0,0,2) -- ++(2,0,0) -- ++(0,0,-2) -- cycle;
  \filldraw[cube] (0,1,0) -- ++(0,0,2) -- ++(2,0,0) -- ++(0,0,-2) -- cycle;
  \filldraw[cube] (2,0,0) -- ++(0,1,0) -- ++(0,0,2) -- ++(0,-1,0) -- cycle;
  \filldraw[cube] (0,0,2) -- ++(0,1,0) -- ++(2,0,0) -- ++(0,-1,0) -- cycle;

  \foreach \i in {0.1,0.3,...,2} {
    \foreach \j in {0.1,0.3,...,1} {
      \pgfmathsetmacro{\x}{\i}
      \pgfmathsetmacro{\y}{\j}
      \pgfmathsetmacro{\dy}{1.0}
      \draw[force, thick] (\x,1,\y) -- ++(0,\dy,0);
    }
  }

  \foreach \i in {0.1,0.3,...,2} {
    \foreach \j in {1.1,1.3,...,2} {
      \pgfmathsetmacro{\x}{\i}
      \pgfmathsetmacro{\y}{\j}
      \pgfmathsetmacro{\dy}{1.0}
      \draw[force, thick] (\x,1,\y) -- ++(0,\dy,0);
    }
  }
\end{tikzpicture}
